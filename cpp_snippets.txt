#include <iostream>
int main() {
    int num;
    std::cin >> num;
    bool isPrime = true;
    for(int i = 2; i < num; i++) {
        if(num % i == 0) {
            isPrime = false;
            break;
        }
    }
    std::cout << (isPrime ? "Prime" : "Not Prime") << std::endl;
    return 0;
}

#include <iostream>
int main() {
    int num;
    std::cin >> num;
    int sum = 0;
    while(num > 0) {
        sum += num % 10;
        num /= 10;
    }
    std::cout << sum << std::endl;
    return 0;
}

#include <iostream>
int main() {
    int a = 10, b = 20;
    std::cout << "Before swap: " << a << " " << b << std::endl;
    int temp = a;
    a = b;
    b = temp;
    std::cout << "After swap: " << a << " " << b << std::endl;
    return 0;
}

#include <iostream>
int main() {
    int n;
    std::cin >> n;
    int reverse = 0;
    while(n > 0) {
        reverse = reverse * 10 + n % 10;
        n /= 10;
    }
    std::cout << reverse << std::endl;
    return 0;
}
